# 🔄 支付系统无缝迁移架构方案

## 📋 方案概述

### 核心设计理念
1. **统一接口抽象**：所有支付方式使用相同的接口
2. **策略模式**：通过配置切换不同的支付实现
3. **回调统一处理**：所有支付回调走同一个处理逻辑
4. **零业务影响**：切换支付方式时，业务代码无需修改

### 架构图
```
前端
  ↓
统一支付接口 (PaymentGateway)
  ↓
支付策略选择器 (PaymentStrategy)
  ├── 虎皮椒支付 (HupijiaoPayment)
  ├── 自建易支付 (SelfHostedPayment)
  ├── 官方支付宝 (OfficialAlipay)
  └── 官方微信 (OfficialWeChat)
  ↓
订单管理器 (OrderManager) ← 保持不变
  ↓
会员激活 (activate_membership) ← 保持不变
```

### 1. 统一支付接口

```python
# backend/payment_gateway.py

from abc import ABC, abstractmethod

class PaymentGateway(ABC):
    """统一支付接口抽象类"""
    
    @abstractmethod
    def create_order(self, order_no, amount, subject, notify_url, return_url=None):
        """
        创建支付订单
        
        Returns:
            {
                'success': True/False,
                'pay_url': '支付链接',
                'qr_code': '二维码内容（可选）',
                'payment_info': {}  # 其他支付信息
            }
        """
        pass
    
    @abstractmethod
    def verify_notify(self, data):
        """验证回调签名"""
        pass
    
    @abstractmethod
    def parse_notify(self, data):
        """解析回调数据"""
        pass
    
    @abstractmethod
    def query_order(self, order_no):
        """查询订单状态"""
        pass
```

### 2. 虎皮椒支付实现

```python
# backend/payment_providers/hupijiao_payment.py

import requests
import hashlib
import time
from payment_gateway import PaymentGateway

class HupijiaoPayment(PaymentGateway):
    """虎皮椒支付实现"""
    
    def __init__(self, merchant_id, api_key, gateway_url):
        self.merchant_id = merchant_id
        self.api_key = api_key
        self.gateway_url = gateway_url
    
    def create_order(self, order_no, amount, subject, notify_url, return_url=None, payment_type='alipay'):
        """创建虎皮椒支付订单"""
        params = {
            'pid': self.merchant_id,
            'type': payment_type,  # alipay/wechat
            'out_trade_no': order_no,
            'notify_url': notify_url,
            'return_url': return_url or '',
            'name': subject,
            'money': str(amount),
            'sign_type': 'MD5',
            'time': str(int(time.time()))
        }
        
        # 生成签名
        sign = self._generate_sign(params)
        params['sign'] = sign
        
        # 发起请求
        try:
            response = requests.post(self.gateway_url, data=params, timeout=10)
            result = response.json()
            
            if result.get('code') == 1:
                return {
                    'success': True,
                    'pay_url': result.get('pay_url'),
                    'qr_code': result.get('qr_code'),
                    'payment_info': result
                }
            else:
                return {
                    'success': False,
                    'error': result.get('msg', '创建订单失败')
                }
        except Exception as e:
            return {
                'success': False,
                'error': f'请求失败: {str(e)}'
            }
    
    def verify_notify(self, data):
        """验证虎皮椒回调签名"""
        sign = data.get('sign', '')
        calculated_sign = self._generate_sign(data)
        return sign == calculated_sign
    
    def parse_notify(self, data):
        """解析虎皮椒回调数据"""
        return {
            'order_no': data.get('out_trade_no'),
            'transaction_id': data.get('trade_no'),
            'status': 'paid' if data.get('trade_status') == 'TRADE_SUCCESS' else 'pending',
            'amount': data.get('money'),
            'payment_data': data
        }
    
    def query_order(self, order_no):
        """查询订单状态"""
        # 虎皮椒查询接口实现
        pass
    
    def _generate_sign(self, params):
        """生成MD5签名"""
        filtered = {k: v for k, v in params.items() if v and k != 'sign'}
        sorted_params = sorted(filtered.items())
        sign_string = '&'.join([f"{k}={v}" for k, v in sorted_params])
        sign_string += f"&key={self.api_key}"
        return hashlib.md5(sign_string.encode('utf-8')).hexdigest()
```

### 4. 支付策略工厂

```python
# backend/payment_factory.py

import os
from payment_providers.hupijiao_payment import HupijiaoPayment
from payment_providers.selfhosted_payment import SelfHostedPayment
from alipay_client import get_alipay_client
from wechat_pay_client import get_wechat_client

class PaymentFactory:
    """支付方式工厂类"""
    
    @staticmethod
    def create_payment_gateway(payment_method='alipay'):
        """
        创建支付网关实例
        
        Args:
            payment_method: 支付方式 alipay/wechat
        
        Returns:
            PaymentGateway实例
        """
        # 从环境变量读取支付提供商配置
        payment_provider = os.getenv('PAYMENT_PROVIDER', 'hupijiao')  # hupijiao/selfhosted/official
        
        if payment_provider == 'hupijiao':
            # 虎皮椒支付
            merchant_id = os.getenv('HUPIJIAO_MERCHANT_ID')
            api_key = os.getenv('HUPIJIAO_API_KEY')
            gateway_url = os.getenv('HUPIJIAO_GATEWAY_URL', 'https://api.hupijiao.com/api/create')
            
            return HupijiaoPayment(merchant_id, api_key, gateway_url)
        
        elif payment_provider == 'selfhosted':
            # 自建易支付系统
            merchant_id = os.getenv('SELFHOSTED_MERCHANT_ID')
            api_key = os.getenv('SELFHOSTED_API_KEY')
            gateway_url = os.getenv('SELFHOSTED_GATEWAY_URL', 'https://pay.yourdomain.com/api/create')
            
            return SelfHostedPayment(merchant_id, api_key, gateway_url)
        
        elif payment_provider == 'official':
            # 官方接口（支付宝/微信）
            if payment_method == 'alipay':
                return get_alipay_client()
            else:
                return get_wechat_client()
        
        else:
            raise ValueError(f"不支持的支付提供商: {payment_provider}")
```

### 5. 统一支付API（修改现有代码）

```python
# backend/payment_api.py (修改后的版本)

from flask import Blueprint, request, jsonify, current_app
from payment_factory import PaymentFactory
from order_manager import get_order_manager
from supabase import create_client

payment_bp = Blueprint('payment', __name__, url_prefix='/api/payment')

@payment_bp.route('/create-order', methods=['POST'])
def create_payment_order():
    """创建支付订单（统一接口）"""
    try:
        data = request.get_json()
        user_id = data.get('user_id')
        plan = data.get('plan')
        payment_method = data.get('payment_method')  # alipay/wechat
        device_info = data.get('device_info', {})
        
        # 验证参数
        if not all([user_id, plan, payment_method]):
            return jsonify({
                'success': False,
                'error': '缺少必要参数'
            }), 400
        
        # 创建订单（这部分逻辑不变）
        supabase = get_supabase_client()
        order_manager = get_order_manager(supabase)
        result = order_manager.create_order(user_id, plan, payment_method, device_info)
        
        if not result['success']:
            return jsonify(result), 400
        
        order = result['order']
        plan_info = result['plan_info']
        
        # 使用工厂创建支付网关（关键：通过配置切换）
        payment_gateway = PaymentFactory.create_payment_gateway(payment_method)
        
        # 创建支付（统一接口，不管底层是虎皮椒还是自建）
        payment_result = payment_gateway.create_order(
            order_no=order['order_no'],
            amount=plan_info['price'] / 100,  # 转换为元
            subject=f"AI背景移除工具 - {plan_info['name']}",
            notify_url=f"{request.host_url}api/payment/notify",
            return_url=f"{request.host_url}payment/success?order_no={order['order_no']}",
            payment_type=payment_method
        )
        
        if payment_result['success']:
            return jsonify({
                'success': True,
                'order': order,
                'payment_info': payment_result
            })
        else:
            return jsonify(payment_result), 400
            
    except Exception as e:
        current_app.logger.error(f"创建支付订单异常: {str(e)}")
        return jsonify({
            'success': False,
            'error': '创建订单失败'
        }), 500

@payment_bp.route('/notify', methods=['POST'])
def payment_notify():
    """统一支付回调接口（所有支付方式都走这里）"""
    try:
        data = request.form.to_dict() or request.get_json() or {}
        
        # 从回调数据判断支付提供商（或从订单查询）
        order_no = data.get('out_trade_no')
        if not order_no:
            return 'FAIL'
        
        # 查询订单获取支付方式
        supabase = get_supabase_client()
        order_manager = get_order_manager(supabase)
        order_result = order_manager.get_order(order_no)
        
        if not order_result['success']:
            return 'FAIL'
        
        order = order_result['order']
        payment_method = order['payment_method']
        
        # 创建对应的支付网关
        payment_gateway = PaymentFactory.create_payment_gateway(payment_method)
        
        # 验证签名
        if not payment_gateway.verify_notify(data):
            current_app.logger.error(f"订单 {order_no} 回调签名验证失败")
            return 'FAIL'
        
        # 解析回调数据
        notify_data = payment_gateway.parse_notify(data)
        
        # 处理订单（这部分逻辑完全不变）
        if notify_data['status'] == 'paid':
            # 更新订单状态
            update_result = order_manager.update_order_status(
                order_no=order_no,
                status='paid',
                transaction_id=notify_data.get('transaction_id'),
                payment_data=notify_data.get('payment_data')
            )
            
            if update_result['success']:
                # 激活会员（这部分逻辑完全不变）
                activate_result = order_manager.activate_membership(order_no)
                if activate_result['success']:
                    current_app.logger.info(f"订单 {order_no} 支付成功，会员已激活")
                else:
                    current_app.logger.error(f"激活会员失败: {activate_result['error']}")
        
        # 返回成功（根据支付方式返回不同格式）
        payment_provider = os.getenv('PAYMENT_PROVIDER', 'hupijiao')
        if payment_provider == 'official' and payment_method == 'wechat':
            return jsonify({'code': 'SUCCESS', 'message': '处理成功'})
        else:
            return 'SUCCESS'
        
    except Exception as e:
        current_app.logger.error(f"支付回调处理异常: {str(e)}")
        return 'FAIL'
```
