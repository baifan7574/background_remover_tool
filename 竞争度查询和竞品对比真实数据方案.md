```python
def compare_competitors_with_google_trends(keywords_list):
    """
    使用Google Trends对比多个关键词的搜索量
    返回：每个关键词的相对搜索量和排名
    """
    # 对比多个关键词
    pytrends.build_payload(keywords_list, cat=0, timeframe='today 12-m')
    interest_over_time = pytrends.interest_over_time()
    
    # 计算每个关键词的平均搜索量
    comparison_data = []
    for keyword in keywords_list:
        avg_volume = interest_over_time[keyword].mean()
        comparison_data.append({
            'keyword': keyword,
            'search_volume': int(avg_volume),
            'rank': 0  # 稍后计算排名
        })
    
    # 按搜索量排序，分配排名
    comparison_data.sort(key=lambda x: x['search_volume'], reverse=True)
    for i, item in enumerate(comparison_data):
        item['rank'] = i + 1
    
    return comparison_data
```
