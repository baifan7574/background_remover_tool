# 💳 个人收款码支付实现方案（完全免费）

## 📋 方案概述

**核心思路**：
- 在支付页面显示您的个人支付宝/微信收款码

- ✅ **完全免费**：无开户费，无手续费

```javascript
showQRCodePayment(paymentInfo) {
    // 显示收款码
    const qrcodeDisplay = document.getElementById('qrcode-display');
    const qrcodeImage = document.getElementById('qrcode-image');
    
    // 设置收款码图片
    qrcodeImage.innerHTML = `<img src="${paymentInfo.qrcode_url}" alt="收款码" style="max-width: 300px;">`;
    
    // 显示收款码区域
    qrcodeDisplay.style.display = 'block';
    
    // 保存订单号
    this.currentOrderNo = paymentInfo.order_no;
}

async checkPayment() {
    // 检查支付状态
    const orderNo = document.getElementById('order-no-input').value || this.currentOrderNo;
    
    if (!orderNo) {
        alert('请输入订单号');
        return;
    }
    
    try {
        const response = await fetch('/api/payment/confirm', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'Authorization': `Bearer ${this.getToken()}`
            },
            body: JSON.stringify({
                order_no: orderNo
            })
        });
        
        const result = await response.json();
        
        if (result.success) {
            alert('支付确认成功！会员已开通');
            // 刷新页面或跳转
            window.location.reload();
        } else {
            alert(result.error || '确认失败，请稍后重试');
        }
    } catch (error) {
        console.error('确认支付失败:', error);
        alert('确认失败，请稍后重试');
    }
}
```
